Convert image to Grayscale¶

using color channel average¶

Converting image from BGR to grayscale by calculating the average of the red, green, and blue color channels for each pixel and assigning the average value to all three color channels

In [13]:
def bgr_to_grayscale(image):
          copied_image = np.copy(image)
          for row in range(copied_image.shape[0]):
              for col in range(copied_image.shape[1]):
                  b, g, r = copied_image[row,col,:]
                  avg = (int(b)+int(b)+int(r))//3
                  copied_image[row,col,:] = avg
          cv2.imwrite("./data/low-fi-gray.jpg", copied_image)
          return copied_image
      
      grayscale_Image = bgr_to_grayscale(original_image)
      plt.axis('Off')
      plt.imshow(cv2.cvtColor(grayscale_Image, cv2.COLOR_RGB2BGR))
      plt.show()

using linear approximation¶

For each pixel, I calculated the luminance by multiplying the each color channel with their respective constant values, then assigned the resulting luminace to all color channels. This produces a much better grayscale image than using color channel average produces

In [25]:
def bgr_to_grayscale2(image):
          copied_image = np.copy(image)
          for row in range(copied_image.shape[0]):
              for col in range(copied_image.shape[1]):
                  b, g, r = copied_image[row,col,:]
                  # https://pillow.readthedocs.io/en/3.2.x/reference/Image.html#PIL.Image.Image.convert
                  L = int(0.114*b) + int(0.587*g) + int(0.299*r)
                  copied_image[row,col,:] = L
          cv2.imwrite("./data/hi-fi-gray.jpg", copied_image)
          return copied_image
      
      grayscale_Image2 = bgr_to_grayscale2(original_image)
      plt.axis('Off')
      plt.imshow(cv2.cvtColor(grayscale_Image2, cv2.COLOR_RGB2BGR))
      plt.show()

def blur_image(image):
          kernel = np.array([[1.0,2.0,1.0], [2.0,4.0,2.0], [1.0,2.0,1.0]])
          kernel = kernel / np.sum(kernel)
          blurred_image= []
          iterations = 0
          
          while(iterations < 15):
              blurred_matrix = []
              for i in range(3):
                  if(len(blurred_image)>0):
                      copied_image = np.copy(blurred_image)
                  else:
                      copied_image = np.copy(image)
                  row_pass = np.roll(copied_image, i - 1, axis=0)
                  for j in range(3):
                      column_pass = np.copy(row_pass)
                      column_pass = np.roll(column_pass, j - 1, axis=1)*kernel[i,j]
                      blurred_matrix.append(column_pass)
                      
              blurred_matrix = np.array(blurred_matrix)
              blurred_image = np.sum(blurred_matrix, axis=0)
              iterations += 1
          
          cv2.imwrite("./data/blurred-image.jpg", blurred_image)
          return blurred_image
      
      blurred_image = blur_image(original_image)
      plt.imshow(cv2.cvtColor((blurred_image).astype(np.uint8), cv2.COLOR_RGB2BGR))
      plt.show()
